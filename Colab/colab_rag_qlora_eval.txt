###CELL### 4：QLoRA 模型验证检测
# ============================================================
# 第一步：彻底清理前面 RAG/BGE 步骤占用的 GPU 和 CPU 内存
# 这是防止 OOM 导致模型被迫卸载到磁盘 → PEFT 无法找到层 的关键步骤！
import gc
import torch

# 清理 Python 所有未引用对象
gc.collect()

# 清理 GPU 显存碎片
if torch.cuda.is_available():
    torch.cuda.empty_cache()
    torch.cuda.ipc_collect()
    print(f"清理完成！当前 GPU 剩余显存: {torch.cuda.memory_reserved() / 1024**3:.1f} GB 已分配")
else:
    print("未检测到 GPU，将使用 CPU（推理速度会慢）")

from transformers import AutoModelForCausalLM, AutoTokenizer
from peft import PeftModel

# 模型加载路径
BASE_MODEL = "deepseek-ai/deepseek-llm-7b-chat"
LORA_PATH = "/content/drive/MyDrive/Graduation_Project/qlora_output"

if not os.path.exists(LORA_PATH):
    print(f"找不到 LoRA 权重路径: {LORA_PATH}。请确认之前是否已经在 Colab 跑过微调并且保存到了 Drive?")
else:
    print("加载 Tokenizer...")
    tokenizer = AutoTokenizer.from_pretrained(BASE_MODEL, trust_remote_code=True)
    
    print("加载半精度 Base Model（float16，全部放进 T4 GPU）...")
    base_model = AutoModelForCausalLM.from_pretrained(
        BASE_MODEL,
        device_map="cuda",        # 明确指定放进 GPU，不走 CPU/磁盘 offload
        dtype=torch.float16,      # 7B float16 约 14GB，T4 15GB 刚好够
        trust_remote_code=True,
    )
    
    print("应用 LoRA 权重...")
    model = PeftModel.from_pretrained(base_model, LORA_PATH)
    model.eval()
    print("模型加载完成！")

    # 测试
    prompts = [
        "User: 请提取以下新闻中的财务指标，输出 JSON 格式。\n内容：苹果公司2025年第一季度营收达到1200亿美元，同比增长5%。\n\nAssistant: "
    ]

    for p in prompts:
        print("\n--- 测试输入 ---")
        print(p)
        inputs = tokenizer(p, return_tensors="pt").to("cuda")
        with torch.no_grad():
            outputs = model.generate(**inputs, max_new_tokens=100)
        res = tokenizer.decode(outputs[0], skip_special_tokens=True)
        print("\n--- 模型输出 ---")
        out_text = res.split("Assistant:")[-1].strip() if "Assistant:" in res else res
        print(out_text)
