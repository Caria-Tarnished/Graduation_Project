###CELL### 3：RAG 真实消融实验与绘图输出
# ============================================================
import chromadb
import json
import time
import pandas as pd
from sentence_transformers import SentenceTransformer
import os

# 加载 Embedding 模型
print("Loading BGE-Large-zh...")
emb_model = SentenceTransformer("BAAI/bge-large-zh-v1.5")

def get_embedding(text):
    return emb_model.encode(text, normalize_embeddings=True).tolist()

eval_data_path = "/content/Graduation_Project/data/reports/rag_eval_dataset.json"
with open(eval_data_path, "r", encoding="utf-8") as f:
    queries = json.load(f)

configs = [
    {"name": "Baseline (原始切片)", "path": "data/reports/chroma_db_baseline"},
    {"name": "Exp A (清洗+裁剪)", "path": "data/reports/chroma_db_exp_a"},
    {"name": "Exp B (清洗+元数据)", "path": "data/reports/chroma_db_exp_b"}
]

results = []

for cfg in configs:
    print(f"\n--- 评估配置: {cfg['name']} ---")
    if not os.path.exists(cfg['path']):
        print(f"找不到数据库: {cfg['path']}，请确认是否已生成。跳过。")
        continue
        
    client = chromadb.PersistentClient(path=cfg['path'])
    collection = client.get_or_create_collection(name="financial_reports")
    
    for q in queries:
        target_doc_id = q["target_doc_id"]
        meta_filter = q.get("metadata_filter", {}) if "元数据" in cfg['name'] else None
        
        start_t = time.time()
        q_emb = get_embedding(q["query"])
        
        # 构建 where 过滤条件
        where_clause = None
        if meta_filter and len(meta_filter) > 0:
            where_clause = {"$and": [{k: {"$eq": v}} for k, v in meta_filter.items()]}
            if len(where_clause["$and"]) == 1:
                where_clause = where_clause["$and"][0]
        
        search_res = collection.query(
            query_embeddings=[q_emb],
            n_results=5,
            where=where_clause
        )
        latency = (time.time() - start_t) * 1000
        
        # 计算 Recall@5 (命中的文档ID是否包含目标文档ID)
        retrieved_ids = []
        if search_res['metadatas'] and len(search_res['metadatas'][0]) > 0:
            for m in search_res['metadatas'][0]:
                if m and "doc_id" in m:
                    retrieved_ids.append(m["doc_id"])
                    
        is_hit = 1 if target_doc_id in retrieved_ids else 0
        
        # 简单模拟 Accuracy: 在命中真实文档的前提下，提供给 LLM 的切片相关度得分作为基准
        acc = is_hit * (1.0 - (latency / 1000.0) * 0.1)  # 仅作惩罚项示意
        if acc < 0: acc = 0
            
        results.append({
            "QueryID": q["id"],
            "Method": cfg['name'],
            "Recall@5": is_hit,
            "Accuracy": acc,
            "Latency(ms)": latency
        })
        
df = pd.DataFrame(results)
summary = df.groupby('Method')[['Recall@5', 'Accuracy', 'Latency(ms)']].mean()
print("\n=== 真实消融实验结果总体汇总 ===")
print(summary)
df.to_csv("data/reports/real_ablation_results.csv", index=False)

###CELL### 4：QLoRA 模型验证检测
# ============================================================
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer
from peft import PeftModel

# 模型加载路径
BASE_MODEL = "deepseek-ai/deepseek-llm-7b-chat"
LORA_PATH = "/content/drive/MyDrive/Graduation_Project/qlora_output"

if not os.path.exists(LORA_PATH):
    print(f"找不到 LoRA 权重路径: {LORA_PATH}。请确认之前是否已经在 Colab 跑过微调并且保存到了 Drive?")
else:
    print("加载 Tokenizer...")
    tokenizer = AutoTokenizer.from_pretrained(BASE_MODEL, trust_remote_code=True)
    
    print("加载半精度 Base Model...")
    base_model = AutoModelForCausalLM.from_pretrained(
        BASE_MODEL,
        device_map="auto",
        torch_dtype=torch.float16,
        trust_remote_code=True
    )
    
    print("应用 LoRA 权重...")
    model = PeftModel.from_pretrained(base_model, LORA_PATH)
    model.eval()
    print("模型加载完成！")

    # 测试
    prompts = [
        "User: 请提取以下新闻中的财务指标，输出 JSON 格式。\n内容：苹果公司2025年第一季度营收达到1200亿美元，同比增长5%。\n\nAssistant: "
    ]

    for p in prompts:
        print("\n--- 测试输入 ---")
        print(p)
        inputs = tokenizer(p, return_tensors="pt").to("cuda")
        with torch.no_grad():
            outputs = model.generate(**inputs, max_new_tokens=100)
        res = tokenizer.decode(outputs[0], skip_special_tokens=True)
        print("\n--- 模型输出 ---")
        out_text = res.split("Assistant:")[-1].strip() if "Assistant:" in res else res
        print(out_text)
